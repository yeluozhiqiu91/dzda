成功消息
fx.succeed({
	title     : '签收成功', 			    // 默认“成功”
	content   : '系统将在3秒后自动返回上级页面。',  // 默认空“”
	delay     : 3000, 			    // 默认0，单位毫秒，n毫秒后执行okey回调
	okeyValue : '返回', 			    // 按钮文字，默认“确定”
	okey      : function() { 		    // 按钮回调
		window.history.back(-1);	
	}
})
